function Library:RegisterConfigFlag(Flag, Type, Default, SetCallback, GetCallback)
	if not Flag or Flag == "" then return end
	Library.ConfigValues[Flag] = {
		Type = Type,
		Value = Default,
		Set = function(Value)
			SetCallback(Value)
			Library.ConfigValues[Flag].Value = GetCallback and GetCallback() or Value
		end,
		Get = GetCallback or function() return Library.ConfigValues[Flag].Value end
	}
end

function Library:SetConfigValue(Flag, Value)
	local Entry = Library.ConfigValues[Flag]
	if Entry and Entry.Set then
		Entry.Set(Value)
	end
end

function Library:GetConfigValue(Flag)
	local Entry = Library.ConfigValues[Flag]
	if Entry and Entry.Get then
		return Entry.Get()
	end
	return nil
end

-- // Functions
local function RoundNumber(x, y)
	return math.round(x * 10 ^ y) / 10 ^ y
end

local function Color3ToHex(Color)
	return string.format("#%02X%02X%02X", math.floor(Color.R * 255), math.floor(Color.G * 255), math.floor(Color.B * 255))
end

local function HexToColor3(Hex)
	Hex = Hex:gsub("#", "")
	if #Hex ~= 6 then return nil end
	local R = tonumber(Hex:sub(1, 2), 16)
	local G = tonumber(Hex:sub(3, 4), 16)
	local B = tonumber(Hex:sub(5, 6), 16)
	if R and G and B then
		return Color3.fromRGB(R, G, B)
	end
	return nil
end

local function Clamp(Value, Min, Max)
	return math.max(Min, math.min(Max, Value))
end

	local function GetViewportSize()
		local Camera = workspace.CurrentCamera
		return Camera and Camera.ViewportSize or Vector2.new(1920, 1080)
	end

	local function ClampToScreen(AbsX, AbsY)
		local ViewportSize = GetViewportSize()
		local Size = Background.AbsoluteSize

		local MinX, MaxX = 0, ViewportSize.X - Size.X
		local MinY, MaxY = 0, ViewportSize.Y - Size.Y

		if MaxX < MinX then MaxX = MinX end
		if MaxY < MinY then MaxY = MinY end

		return math.clamp(AbsX, MinX, MaxX), math.clamp(AbsY, MinY, MaxY)
	end

	local function BeginDrag(InputPosition)
		Dragging = true
		DragStart = InputPosition
		StartPosition = Background.Position
		GoalPosition = Background.Position
	end

	local function UpdateDrag(InputPosition)
		if not Dragging or not StartPosition then return end

		local Delta = InputPosition - DragStart
		local ViewportSize = GetViewportSize()
		local AnchorOffsetX = StartPosition.X.Scale * ViewportSize.X + StartPosition.X.Offset
		local AnchorOffsetY = StartPosition.Y.Scale * ViewportSize.Y + StartPosition.Y.Offset

		local TargetAbsX = AnchorOffsetX + Delta.X - (Background.AbsoluteSize.X * Background.AnchorPoint.X)
		local TargetAbsY = AnchorOffsetY + Delta.Y - (Background.AbsoluteSize.Y * Background.AnchorPoint.Y)

		local ClampedX, ClampedY = ClampToScreen(TargetAbsX, TargetAbsY)

		local FinalX = ClampedX + (Background.AbsoluteSize.X * Background.AnchorPoint.X)
		local FinalY = ClampedY + (Background.AbsoluteSize.Y * Background.AnchorPoint.Y)

		GoalPosition = UDim2.new(0, FinalX, 0, FinalY)
	end

	local function EndDrag()
		Dragging = false
		DragStart = nil
		StartPosition = nil
	end

	local function Lerp(a, b, m)
		return a + (b - a) * m
	end

				SliderBarUIGradient.Parent = SliderBar

				local Sliding = false

				local function SetSliderVisual(Value, Tweened)
					local Alpha = math.clamp((Value - SliderOptions.Min) / (SliderOptions.Max - SliderOptions.Min), 0, 1)

					local BoxGoal = UDim2.new(Alpha, 0, 0.5, 0)
					local BarGoal = UDim2.new(Alpha, 0, 1, 0)

					if Tweened then
						TweenService:Create(SliderBox, SliderTweenInfo, {Position = BoxGoal}):Play()
						TweenService:Create(SliderBar, SliderTweenInfo, {Size = BarGoal}):Play()
					else
						SliderBox.Position = BoxGoal
						SliderBar.Size = BarGoal
					end

					SliderValue.Text = tostring(Value)
				end

				local function SetSlider(Value)
					Value = math.clamp(RoundNumber(Value, SliderOptions.Rounding), SliderOptions.Min, SliderOptions.Max)
					SetSliderVisual(Value, true)
					SliderOptions.Callback(Value)
				end

					Sliding = true
				end)

					if Input.UserInputType == Enum.UserInputType.MouseButton1 then
						if Sliding then
							Sliding = false
							local Text = SliderValue.Text
							if tonumber(Text) then
								SetSlider(tonumber(Text))
							end
						end
					end
				end)

				UserInputService.InputChanged:Connect(function(Input, GameProcessed)
					if Sliding and not GameProcessed and Input.UserInputType == Enum.UserInputType.MouseMovement then
						local MouseOffset = UserInputService:GetMouseLocation().X - SliderLine.AbsolutePosition.X
						local Percentage = math.clamp(MouseOffset / SliderLine.AbsoluteSize.X, 0, 1)
						local Value = RoundNumber(SliderOptions.Min + (SliderOptions.Max - SliderOptions.Min) * Percentage, SliderOptions.Rounding)

						SetSliderVisual(Value, false)
						SliderOptions.Callback(Value)
					end
				end)

				SliderValue.FocusLost:Connect(function(EnterPressed)
					if EnterPressed then
						local Text = SliderValue.Text

						if tonumber(Text) then
							SetSlider(tonumber(Text))
						end
					end
				end)

					Library:RegisterConfigFlag(SliderOptions.Flag, "slider", SliderOptions.Default, SetSlider, function() return tonumber(SliderValue.Text) or SliderOptions.Default end)
				end
			end

				LayoutOrder += 1

				local DropdownHolder = Instance.new("Frame")
				DropdownHolder.BackgroundColor3 = Color3.fromRGB(27, 30, 37)
				DropdownHolder.Position = UDim2.new(1, -96, 0, 3)
				DropdownHolder.Size = UDim2.new(0, 96, 0, 18)
				DropdownHolder.ZIndex = 4
				DropdownHolder.ClipsDescendants = true
				DropdownHolder.Name = "DropdownHolder"
				DropdownHolder.Parent = DropdownFrame

				local DropdownHolderUICorner = Instance.new("UICorner")
				DropdownHolderUICorner.CornerRadius = UDim.new(0, 4)
				DropdownHolderUICorner.Parent = DropdownHolder

				local DropdownHolderUIListLayout = Instance.new("UIListLayout")
				DropdownHolderUIListLayout.SortOrder = Enum.SortOrder.LayoutOrder
				DropdownHolderUIListLayout.Parent = DropdownHolder

				local IsDropdownClosed = true
				local SelectedOption = DropdownOptions.Default

				local function SetDropdown(Value)
					if not table.find(DropdownOptions.Options, Value) then return end
					SelectedOption = Value
					DropdownOptions.Callback(Value)

					for _, DropdownOptionName2 in DropdownOptions.Options do
						if DropdownOptionName2 == Value then
							DropdownHolder[DropdownOptionName2].LayoutOrder = 0
						else
							DropdownHolder[DropdownOptionName2].LayoutOrder = 1
						end
					end
				end

				local DropdownLayoutOrder = 0
				for _, DropdownOptionName in DropdownOptions.Options do
					DropdownLayoutOrder += 1

					local DropdownOptionButton = Instance.new("TextButton")
					DropdownOptionButton.BackgroundTransparency = 1
					DropdownOptionButton.Size = UDim2.new(1, 0, 0, 18)
					DropdownOptionButton.ZIndex = 5
					DropdownOptionButton.FontFace = Font.new("rbxasset://fonts/families/Montserrat.json", Enum.FontWeight.Regular, Enum.FontStyle.Normal)
					DropdownOptionButton.Text = DropdownOptionName
					DropdownOptionButton.TextColor3 = Color3.fromRGB(84, 97, 119)
					DropdownOptionButton.TextScaled = true
					DropdownOptionButton.AutoButtonColor = false
					DropdownOptionButton.Name = DropdownOptionName
					DropdownOptionButton.Parent = DropdownHolder

					if DropdownOptionName == DropdownOptions.Default then
						DropdownOptionButton.LayoutOrder = 0
					else
						DropdownOptionButton.LayoutOrder = 1
					end

					DropdownOptionButton.MouseButton1Click:Connect(function()
						if IsDropdownClosed then
							IsDropdownClosed = false
							Library.OpenDropdowns += 2
							TweenService:Create(DropdownHolder, TweenInformation, {Size = UDim2.new(0, 96, 0, 18 * #DropdownOptions.Options)}):Play()
							DropdownHolder.ZIndex = 999 + Library.OpenDropdowns

							for _, DropdownOptionName2 in DropdownOptions.Options do
								DropdownHolder[DropdownOptionName2].ZIndex = 1000 + Library.OpenDropdowns
							end
						else
							IsDropdownClosed = true
							Library.OpenDropdowns -= 2
							local Tween = TweenService:Create(DropdownHolder, TweenInformation, {Size = UDim2.new(0, 96, 0, 18)})
							Tween:Play()

							SetDropdown(DropdownOptionName)

							Tween.Completed:Connect(function()
								DropdownHolder.ZIndex = 4

								for _, DropdownOptionName2 in DropdownOptions.Options do
									DropdownHolder[DropdownOptionName2].ZIndex = 5
								end
							end)
						end
					end)
				end

				if DropdownOptions.Flag then
					Library:RegisterConfigFlag(DropdownOptions.Flag, "dropdown", SelectedOption, SetDropdown, function() return SelectedOption end)
				end
			end
